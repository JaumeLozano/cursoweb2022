SELECT NOMBREARTÍCULO, SECCIÓN, PRECIO, PRECIO*0.07 AS 7PORCIENTO,  ROUND(PRECIO-(PRECIO*0.07), 2) AS DESCUENTO_7 FROM PRODUCTOS;
2. 
SELECT FECHA,SECCIÓN, NOMBREARTÍCULO, PRECIO, ROUND(PRECIO-2,2) AS DTO_2€_EN_CERAMICA FROM PRODUCTOS WHERE SECCIÓN='CERÁMICA' ORDER BY FECHA DESC; 
3.
SELECT NOMBREARTÍCULO, SECCIÓN, PRECIO,  ROUND(PRECIO*1.02, 2) AS PRECIO_AUMENTADO_EN_2 FROM PRODUCTOS WHERE SECCIÓN='FERRETERÍA';

//PRECIO+PRECIO*0.02 = PRECIO*1.02

4. 
SELECT NOMBREARTÍCULO, SECCIÓN, PRECIO, FECHA, PRECIO+10 AS SUMA10, PRECIO*1.10 AS INCREMENTO10 FROM PRODUCTOS WHERE SECCIÓN='DEPORTES' OR SECCIÓN='CERÁMICA' ORDER BY SECCIÓN, PRECIO
